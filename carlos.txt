#ifndef FILE_CPP
#define FILE_CPP

#include <fstream>

using namespace std;

class InputFile {
    private:
        ifstream file;
    
    public:
        InputFile(string filename) {
            file = ifstream(filename);
        }

        void close() {
            file.close();
        }

        istream& readLine(string &line) {
            return getline(file, line);
        }
};

class OutputFile {
    protected:
        string filename;
        ofstream file;

    public:
        OutputFile(string _filename) {
            filename = _filename;
            file = ofstream(filename);
        }

        void write(string line) {
            file << line;
        }

        void close() {
            file.close();
        }

        string getFilename() {
            return filename;
        }
};
